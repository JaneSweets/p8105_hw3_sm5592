``` r
demographics_df =
  read_csv("data/nhanes_covar.csv", skip = 4) |> # skip the first 4 rows
  janitor::clean_names() |> 
  drop_na() |> # exclude participants with missing demographic data
  filter(age >= 21) |> #include only those 21 or more than 21 years old
```
